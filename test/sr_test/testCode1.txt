(* program to find bigger of two numbers TEST of if-else statement*)
MODULE Pattern;
	VAR a, b, test : INTEGER ;
	PROCEDURE bigger(a, b: INTEGER) : INTEGER;
	BEGIN 
		IF  ( a > b ) THEN
			RETURN a
		ELSIF (a = b) THEN
			RETURN 0
		ELSE
			RETURN b
		END ;
	END bigger;

BEGIN
	a := 1;
	b := 1;
	test := bigger(a,b);
	WRITEINT(test);
END Pattern.
