(*program to test call be reference of a pointer *)
MODULE Pattern;
	VAR x, y : POINTER TO INTEGER ;
	PROCEDURE func(VAR x, y : POINTER TO INTEGER);
	BEGIN 
		x^:=2;
		y^:=1;
	END func;

BEGIN
	NEW(x);
	NEW(y);
	x^:=1;
	y^:=2;
	WRITEINT(x^);
	WRITEINT(y^);
	func(x,y);
	WRITEINT(x^);
	WRITEINT(y^);
END Pattern.
