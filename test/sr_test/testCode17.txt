(* Program for a 3 level code *)
MODULE Pattern;
	VAR x : INTEGER;
	PROCEDURE func(VAR x : INTEGER);
		VAR y: INTEGER;
		PROCEDURE prod(x, y :INTEGER) : INTEGER ;
		BEGIN
			RETURN x*y
		END prod;
	BEGIN 
		IF x < 0 THEN
			x := -1*x;
		END;
		y:=1;
		WHILE y <= 10 DO
			WRITEINT(prod(x, y));
			y := y +1 ;
		END;
	END func;

BEGIN
	x:=7;
	func(x);
END Pattern.
