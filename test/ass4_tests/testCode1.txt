MODULE testCode1;

VAR a, bh, test : INTEGER ;
 
PROCEDURE bigger(a, bh: INTEGER) : INTEGER;
	BEGIN
		IF  ( a > bh ) THEN
			RETURN a ;
		ELSEIF (a = bh) THEN
			RETURN 0 ;
		ELSE
			RETURN bh ;
		END ;
	END bigger;

 BEGIN
 	READINT(a);
 	READINT(bh);
 	WRITEINT(a);
 	WRITEINT(bh);
	test := bigger(a,bh);
 	WRITEINT(test);
 END testCode1.
