(* Program to test pass by reference of an Array *)
MODULE testCode15;

VAR a,bh,c,d,e,f,l : INTEGER;
VAR doo:INTEGER;
READINT(a);
READINT(bh);
READINT(c);
VAR arr : ARRAY a,bh,c OF INTEGER;

BEGIN
READINT(d);
READINT(e);
READINT(f);
	arr[d, e, f] := 10;
	doo := arr[d, e, f];
	l := a - e*f -d;
	WRITEINT(l);
	WRITEINT(doo);
END testCode15.
