(* WHILE DO statement *)
MODULE Pattern;
	VAR x : INTEGER;
	PROCEDURE func(x : INTEGER);
		VAR y: INTEGER;
		PROCEDURE prod(x, y :INTEGER) : INTEGER ;
		BEGIN
			RETURN x*y
		END prod;
	BEGIN
		IF x < 0 THEN
			x := -1*x;
		END;
		y:=1;
		DO
		BEGIN
			WRITEINT(prod(x, y));
			y := y +1 ;
		END WHILE(y>10) ;
	END func;

BEGIN
	x:=7;
	func(x);
END Pattern.
